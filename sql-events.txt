CREATE TABLE venues (
	id INTEGER PRIMARY KEY,

	name TEXT,
	address TEXT,
	postcode TEXT,
	info TEXT
);

CREATE TABLE events (
	id INTEGER PRIMARY KEY,

	/* Event data */
	title TEXT,
	date DATETIME,
	location TEXT,	/* Either text or numeric reference to venues */
	blurb TEXT,
	url TEXT,
	type TEXT,		/* ATM this can only be "film" */
	cost TEXT,		/* NULL = free */

	/* State crap */
	state TEXT,		/* "submitted", "validated", "approved" */
	email TEXT,		/* Delete after submitted? */
	key TEXT			/* key needed for validation */
);

CREATE TABLE users (
	email TEXT PRIMARY KEY,
	banned BOOLEAN,
	salt TEXT,
	digest TEXT
);
