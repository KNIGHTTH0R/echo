CREATE TABLE events
(
	id INT auto_increment PRIMARY KEY,
	
	/* actual event data */
	title TEXT,
	date DATETIME,
	location TEXT,
	blurb TEXT,
	facebook_id INT,
	url_info TEXT,
	
	/* maint stuff */
	submitter_email TEXT,
	approved BOOLEAN
);

CREATE TABLE users
(
	email TEXT PRIMARY KEY,
	validated BOOLEAN,
	approved BOOLEAN,
	banned BOOLEAN
);
