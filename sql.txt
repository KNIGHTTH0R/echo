CREATE TABLE venues
{
	name TEXT PRIMARY KEY,
	address TEXT,
	info TEXT
};

CREATE TABLE events
(
	id INTEGER PRIMARY KEY,

	/* Event data */
	title TEXT,
	date DATETIME,
	location TEXT,
	blurb TEXT,
	url TEXT,
	type TEXT, /* ATM this can only be "film" */
	free INTEGER	/* 1 = free, 0 = not free */

	/* State crap */
	state TEXT,		/* "submitted", "validated", "approved" */
	email TEXT,		/* Delete after submitted? */
	key TEXT,		/* key needed for validation */
);

CREATE TABLE users
(
	email TEXT PRIMARY KEY,
	banned BOOLEAN,
	salt TEXT,
	digest TEXT
);

-- 

CREATE TABLE posts (
	id TEXT PRIMARY KEY,
	title TEXT,
	link TEXT,
	date DATETIME,
	summary TEXT,
	image TEXT,	 /* this is an URL... maybe make it a data: link? */
	feed_url TEXT
);

CREATE TABLE feeds (
	feed_url TEXT PRIMARY KEY,
	site_url TEXT,
	title TEXT
);

CREATE VIEW post_info AS SELECT * FROM posts LEFT OUTER JOIN feeds ON posts.feed_url = feeds.feed_url;
